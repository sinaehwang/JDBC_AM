DROP DATABASE IF EXISTS article_manager;

CREATE TABLE article(
id INT(10) UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,

updateDate DATETIME NOT NULL,

`body` TEXT NOT NULL
);

CREATE TABLE `member` (
